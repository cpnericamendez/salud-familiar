// ───────────────────────────────────────────────────────────── // Firebase Cloud Function — Recordatorios diarios de turnos // Se ejecuta todos los días a las 8:00 AM (Argentina, UTC-3) // ───────────────────────────────────────────────────────────── const functions = require("firebase-functions"); const admin = require("firebase-admin"); admin.initializeApp(); exports.recordatorioTurnos = functions.pubsub .schedule("0 11 * * *") // 11 UTC = 8am Argentina .timeZone("America/Argentina/Buenos_Aires") .onRun(async () => { const db = admin.firestore(); const messaging = admin.messaging(); // Get family data const snap = await db.doc("familyData/shared").get(); if (!snap.exists) return null; const data = snap.data(); // Get all FCM tokens const tokensSnap = await db.collection("fcmTokens").get(); const tokens = tokensSnap.docs.map(d => d.data().token).filter(Boolean); if (!tokens.length) return null; // Find today and tomorrow appointments const today = new Date(); today.setHours(0,0,0,0); const tomorrow = new Date(today); tomorrow.setDate(tomorrow.getDate()+1); const toISO = d => d.toISOString().split("T")[0]; const todayStr = toISO(today); const tmrwStr = toISO(tomorrow); const members = data.members || []; const appointments = data.appointments || []; const messages = []; appointments.forEach(a => { const mb = members.find(m => m.id === a.memberId); const name = mb?.name || "Integrante"; if (a.date === todayStr) { messages.push({ title: `🔴 Turno HOY — ${name}`, body: `${a.title}${a.time ? " a las " + a.time : ""}${a.specialist ? " · " + a.specialist : ""}` }); } else if (a.date === tmrwStr) { messages.push({ title: `🟡 Turno MAÑANA — ${name}`, body: `${a.title}${a.time ? " a las " + a.time : ""}${a.specialist ? " · " + a.specialist : ""}` }); } }); // Also check pending vaccines due today/tomorrow // (simplified — full logic in frontend) if (!messages.length) return null; // Send one notification per appointment to all tokens for (const msg of messages) { await messaging.sendEachForMulticast({ tokens, notification: { title: msg.title, body: msg.body }, webpush: { notification: { icon: "/icon-192.png", badge: "/icon-192.png", vibrate: [200,100,200] }, fcmOptions: { link: "/" } } }); } console.log(`Sent ${messages.length} reminders to ${tokens.length} devices`); return null; });
